1.2 t2.py 

from pythonqt import *

a = Application()
quit = PushButton("Quit")
quit.resize(75, 30)
quit.setFont(QFontV("times", 18, QFont.Bold))
quit.connect("clicked", a.quit)
a.setMainWidget(quit)
quit.show()
a.execLoop()
